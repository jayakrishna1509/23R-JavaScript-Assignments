for,do-while and while loops iterating array of objects in JavaScript 


    // for loop
    var aos = ["mom", "js", "malayalam", "html", "css", "dad"];
    var op = [];
    var j = 0;
    var words = ["mom", "dad", "malayalam"]; //order
    for (var i = 0; i < words.length; i++) {
      for (var k = 0; k < aos.length; k++) {
        if (aos[k] === words[i]) {
          op[j] = aos[k];
          j++;
          break;
        }
      }
    }
    console.log(op);
    
    // while loop
    var aos = ["mom", "js", "malayalam", "html", "css", "dad"];
    var op = [];
    var i = 0;
    var j = 0;
    var words = ["mom", "dad", "malayalam"]; //order
    while (i < words.length) {
      for (var k = 0; k < aos.length; k++) {
        if (aos[k] === words[i]) {
          op[j] = aos[k];
          j++;
          break;
        }
      }
      i++;
    }
    console.log(op);
    
    // do-while loop
    var aos = ["mom", "js", "malayalam", "html", "css", "dad"];
    var op = [];
    var i = 0;
    var j = 0;
    var words = ["mom", "dad", "malayalam"]; //order
    do {
      for (var k = 0; k < aos.length; k++) {
        if (aos[k] === words[i]) {
          op[j] = aos[k];
          j++;
          break;
        }
      }
      i++;
    } while (i < words.length);
    console.log(op);
